⚙ Datenbank Setup

Einmalig im Supabase SQL Editor ausführen. Füge auch die neue Spalte module_type hinzu falls du das Tool aktualisierst:

-- Neue Tabellen erstellen (falls noch nicht vorhanden) CREATE TABLE IF NOT EXISTS technologies ( id BIGSERIAL PRIMARY KEY, name TEXT NOT NULL UNIQUE, branch TEXT NOT NULL, parent_name TEXT, found_in_parent_level INTEGER, description TEXT, tech_class TEXT, created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS tech_modules ( id BIGSERIAL PRIMARY KEY, tech_name TEXT NOT NULL REFERENCES technologies(name) ON DELETE CASCADE, module_name TEXT NOT NULL, unlock_level INTEGER NOT NULL, module_type TEXT DEFAULT 'bauteil' ); CREATE TABLE IF NOT EXISTS tech_bonuses ( id BIGSERIAL PRIMARY KEY, tech_name TEXT NOT NULL REFERENCES technologies(name) ON DELETE CASCADE, bonus_text TEXT NOT NULL ); -- Falls tech_modules bereits existiert, Spalte ergänzen: ALTER TABLE tech_modules ADD COLUMN IF NOT EXISTS module_type TEXT DEFAULT 'bauteil'; -- RLS ALTER TABLE technologies ENABLE ROW LEVEL SECURITY; ALTER TABLE tech_modules ENABLE ROW LEVEL SECURITY; ALTER TABLE tech_bonuses ENABLE ROW LEVEL SECURITY; CREATE POLICY "r" ON technologies FOR SELECT USING (true); CREATE POLICY "i" ON technologies FOR INSERT WITH CHECK (true); CREATE POLICY "u" ON technologies FOR UPDATE USING (true); CREATE POLICY "r" ON tech_modules FOR SELECT USING (true); CREATE POLICY "i" ON tech_modules FOR INSERT WITH CHECK (true); CREATE POLICY "u" ON tech_modules FOR UPDATE USING (true); CREATE POLICY "d" ON tech_modules FOR DELETE USING (true); CREATE POLICY "r" ON tech_bonuses FOR SELECT USING (true); CREATE POLICY "i" ON tech_bonuses FOR INSERT WITH CHECK (true); CREATE POLICY "u" ON tech_bonuses FOR UPDATE USING (true); CREATE POLICY "d" ON tech_bonuses FOR DELETE USING (true);
